Abstract

This is a working example of sending button clicks and text to the simple built-in Calculator and Notepad applications.

You will need “Class” and “Caption” information about the GUIs that you are working with. You can get this information from a wonderful Open Source application called WinSpy++ by james@catch22.net.

http://www.catch22.net/software/winspy

Function testwithCalc()

'This enters "1+2=" into the calculator and then displays "3." in a Message Box.

Dim hWindow As Long
Dim hThread As Long
Dim hProcess As Long
Dim lProcessId As Long
Dim lngResult As Long
Dim lngReturnValue As Long
Dim childhandle As Long

'open the calc
Shell "calc.exe", vbNormalNoFocus

'get the handle of the application by the title bar text
hWindow = FindWindow(vbNullString, "Calculator")

'Get the handle of the Child window of it's class and caption
childhandle = FindWindowEx(hWindow, 0, "Button", "1")

'send a message to that child window
lngReturnValue = PostMessage(childhandle, BM_CLICK, 0&, 0&)

'and repeat with other child windows

childhandle = FindWindowEx(hWindow, 0, "Button", "+")
lngReturnValue = PostMessage(childhandle, BM_CLICK, 0&, 0&)

childhandle = FindWindowEx(hWindow, 0, "Button", "2")
lngReturnValue = PostMessage(childhandle, BM_CLICK, 0&, 0&)

childhandle = FindWindowEx(hWindow, 0, "Button", "=")
lngReturnValue = PostMessage(childhandle, BM_CLICK, 0&, 0&)

'get the handle of the 'output' child window
childhandle = FindWindowEx(hWindow, 0, "Edit", "")

'get the text from 'output' child window and show it in a messagebox
MsgBox GetText(childhandle)

'lngReturnValue = PostMessage(hWindow, WM_CLOSE, 0&, 0&)

End Function

Function testWithNotepad()

'This code gets text from Notepad and then replaces it with a other text.

Dim hWindow As Long
Dim childhandle As Long

'get the window handle on a window by the title
hWindow = FindWindow(vbNullString, "untitled - Notepad")

'stop if it is missing
If hWindow = 0 Then
 MsgBox "for this demo you need a 'untitled - Notepad' window open."
 Exit Function
End If

'get the handle of the 'edit' class child window
childhandle = FindWindowEx(hWindow, 0, "Edit", "")

'show the text from child window
MsgBox GetText(childhandle)

'send text to the child window
MsgBox SendMessage(childhandle, WM_SETTEXT, ByVal CLng(0), ByVal "from vbaCode")

End Function
''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''

Public Function GetText(hwnd As Long) As String
Dim strTemp As String
Dim lngLength As Long

lngLength = SendMessage(hwnd, WM_GETTEXTLENGTH, ByVal 0, ByVal 0)
strTemp = Space(lngLength)
SendMessage hwnd, WM_GETTEXT, ByVal lngLength + 1, ByVal strTemp

GetText = strTemp
End Function
