These are my favorite jQuery functions.

 Using the ajaxStart and ajaxStop event to display and clear a “LOADING…” message
 $(document).ajaxStop(function() {

        document.getElementById(‘loading’).innerHTML = ”

})//end document).ajaxStop event

$(document).ajaxStart(function() {

      document.getElementById(‘loading’).innerHTML = ‘<b>L O A D I N G . . .</b>’

})//end document).ajaxStart event
“Ajax” an ASP file into a div element with a callback using the wonderful LOAD function
 $(‘#bydollar’).load(‘bydollar.asp’+$(‘#hiddentext’).val()),
               function() {
               alert(‘hiddentext is {‘ + $(‘#hiddentext’).val() + ‘}’);
               alert(‘Load bydollar.asp was performed.’);
               };


 Posting to an ASP file from an HTML file with the jQuery Post function
        $.post(“insert.asp”, {
           uid: document.getElementById(‘uid’).value,
           ContactName: document.getElementById(‘ContactName’).value,
           Notes: document.getElementById(‘ta2’).value,
           ContactTitle: document.getElementById(‘ContactTitle’).value,
           FollowUpDate: document.getElementById(‘FollowUpDate’).value,
           ContactPhone: document.getElementById(‘ContactPhone’).value,
           ContactEmail: document.getElementById(‘ContactEmail’).value,
           invoice: document.getElementById(‘invoice’).value,
           companyid: document.getElementById(‘companyid’).value,
           companyname: document.getElementById(‘companyname’).value
       }, //end post function
       function(data){
           alert(‘Note Posted’);
           //overwrite input area
           document.getElementById(‘uid’).value =””
           document.getElementById(‘ta2’).value =””
           document.getElementById(‘ContactEmail’).value =””
           document.getElementById(‘inputarea’).innerHTML = data

       }); //end call back

    }// end post

$(‘tr’).mouseover(function(){
    $(this).addClass(‘over’);
});
$(‘tr’).mouseout(function(){
    $(this).removeClass(‘over’);
});
</script>
